### 4.1 Huidige Situatie


### 4.2 Verbeteringen

#### A. Vector Embeddings voor Semantic Search
```sql
-- Nieuwe kolom voor embeddings
ALTER TABLE "Article" ADD COLUMN "embedding" vector(1536);

-- Index voor snelle similarity search
CREATE INDEX ON "Article" USING ivfflat (embedding vector_cosine_ops);
```

#### B. Knowledge Chunks Systeem
```prisma
model KnowledgeChunk {
  id            String   @id @default(cuid())
  articleId     String?
  article       Article? @relation(fields: [articleId], references: [id])

  // Content
  content       String   @db.Text
  summary       String?

  // Metadata
  source        String   // "onderzoek", "artikel", "faq"
  sourceRef     String?  // PDF naam of URL
  topic         String   // Hoofdonderwerp
  subtopics     String[] // Tags

  // Vector
  embedding     Unsupported("vector(1536)")?

  // Quality
  confidence    Float    @default(1.0)
  lastVerified  DateTime @default(now())

  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt
}
```

#### E. RAG Pipeline

```typescript
async function getIrisResponse(message: string, sessionId: string) {
  // 1. Embed user message
  const queryEmbedding = await embed(message);

  // 2. Hybrid search: semantic + keyword
  const relevantChunks = await prisma.$queryRaw`
    SELECT id, content, source, "sourceRef", topic,
           1 - (embedding <=> ${queryEmbedding}::vector) as similarity
    FROM "KnowledgeChunk"
    WHERE 1 - (embedding <=> ${queryEmbedding}::vector) > 0.7
    ORDER BY similarity DESC
    LIMIT 5
  `;

  // 3. Get related articles
  const articles = await getRelatedArticles(relevantChunks);

  // 4. Build context
  const context = buildContext(relevantChunks, articles);

  // 5. Generate response
  return await generateResponse(message, context, sessionId);
}
```

### 5.1 Nieuwe Modellen

```prisma
// Onderzoeksbronnen
model ResearchSource {
  id            String   @id @default(cuid())
  filename      String   @unique
  title         String

  // Metadata
  topic         String
  subtopics     String[]
  publicationYear Int?
  authors       String[]

  // Content
  summary       String?  @db.Text
  keyFindings   String[]

  // Processing
  processedAt   DateTime?
  chunkCount    Int      @default(0)

  // Relaties
  chunks        KnowledgeChunk[]

  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt
}

// Kennischunks voor RAG
model KnowledgeChunk {
  id            String   @id @default(cuid())

  // Bron
  sourceType    ChunkSourceType
  articleId     String?
  article       Article? @relation(fields: [articleId], references: [id])
  researchId    String?
  research      ResearchSource? @relation(fields: [researchId], references: [id])

  // Content
  content       String   @db.Text
  summary       String?

  // Categorisatie
  topic         String
  subtopics     String[]

  // Vector embedding (via pgvector)
  // embedding  vector(1536) - added via migration

  // Quality
  confidence    Float    @default(1.0)
  usageCount    Int      @default(0)

  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt

  @@index([topic])
  @@index([sourceType])
}

enum ChunkSourceType {
  RESEARCH
  ARTICLE
  FAQ
  MANUAL
}

// Artikel uitbreiding
model Article {
  // ... bestaande velden ...

  // Nieuwe velden
  pillarPage    Boolean  @default(false)
  cluster       String?  // "retentie", "impact", etc.
  researchRefs  String[] // PDF referenties

  // SEO uitbreiding
  focusKeyword  String?
  seoScore      Int?     // 0-100

  // Relaties
  chunks        KnowledgeChunk[]
}
```
